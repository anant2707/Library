class Electricity {
    int units; double amount;
    Electricity(int u) { units = u; }
    double Bill() {
        if(units<=100)       amount = units * 0.50;
        else if(units<=300)  amount = 50 + (units-100)*0.75;
        else                 amount = 50 + 150 + (units-300)*0.60;
        return amount;
    }
}

    More_Electricity(int u) { super(u); }
    @Override
    double Bill() {

            double sc = (base-250)*0.15;
            System.out.println("Surcharge: Rs."+sc);
            System.out.println("Total: Rs."+(base+sc));
            return base+sc;
        }
        System.out.println("Total: Rs."+base); return base;
    }
}

Write a menu-driven program using do-while + switch. Menu options: 1.Factorial  2.Palindrome  3.ASCII of character  4.Exit. Keep running until user selects Exit.

import java.util.Scanner;
public class MenuDriven {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int ch;
        do {
            System.out.println("\n1.Factorial 2.Palindrome 3.ASCII 4.Exit");
            System.out.print("Choice: "); ch = sc.nextInt();
            switch(ch) {
                case 1: System.out.print("n: "); int n=sc.nextInt();
                        long f=1; for(int i=1;i<=n;i++) f*=i;
                        System.out.println(n+"! = "+f); break;
                case 2: System.out.print("Number: "); int num=sc.nextInt();
                        int orig=num,rev=0;
                        while(num>0){rev=rev*10+num%10;num/=10;}
                        System.out.println(orig+(orig==rev?" is":" is NOT")+" palindrome"); break;
                case 3: System.out.print("Char: "); char c=sc.next().charAt(0);
                        System.out.println("ASCII of '"+c+"' = "+(int)c); break;
                case 4: System.out.println("Exiting."); break;
                default: System.out.println("Invalid!");
            }
        } while(ch!=4);
        sc.close();
    }
}

class InsufficientFundsException extends Exception {

Develop Java program that takes ArrayList of integers and returns sum of all elements. Traverse using Iterator.


import java.util.*;
public class ArrayListSum {
    static int sumOfList(ArrayList<Integer> list) {
        int sum=0; for(int n:list) sum+=n; return sum;
    }
    public static void main(String[] args) {
        ArrayList<Integer> nums = new ArrayList<>(Arrays.asList(10,20,35,45,50));
        System.out.println("List : " + nums);
        System.out.println("Sum  : " + sumOfList(nums));
        System.out.print("Iterator: ");
        Iterator<Integer> it = nums.iterator();
        while(it.hasNext()) System.out.print(it.next()+" ");
    }
}

public class LinkedListDemo {

        System.out.println("After removeFirst: " + ll);
        // For-each
        System.out.print("For-each: ");
        for(String s : ll) System.out.print(s+" ");
